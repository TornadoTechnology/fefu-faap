    std::cout << "Total time: ";

    int k;

```cpp
#include<iostream>

int main() {
    std::cout << "Total money: ";

    int money;
    std::cin >> money;

    int amount5000 = money / 5000;
    money %= 5000;

    int amount1000 = money / 1000;
    money %= 1000;

    int amount500 = money / 500;
    money %= 500;

    int amount100 = money / 100;
    money %= 100;

    std::cout
        << amount5000 << " - 5000, "
        << amount1000 << " - 1000, "
        << amount500  << " - 500, "
        << amount100  << " - 100, "
        << money      << " - 1"
        << std::endl;

    return 0;
}
```
